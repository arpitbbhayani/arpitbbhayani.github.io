int arr[100010];

void reverse(int start, int end) {
    while(start < end) {
        int t = arr[start];
        arr[start] = arr[end];
        arr[end] = t;
        start++; end--;
    }
}

int main(int argc, char *argv[]) {
    int n, k;
    scanf("%d%d", &n, &k);
    for(int i = 0 ; i < n ; i ++) {
        scanf("%d", &arr[i]);
    }

    reverse(0, n-1);
    reverse(0, n-k-1);
    reverse(n-k, n-1);

    for(int i = 0 ; i < n ; i ++) {
        printf("%d ", arr[i]);
    }
    printf("\n");

    return 0;
}
