int arr[101];

int main(int argc, char *argv[]) {
    int n, k;
    scanf("%d%d", &n, &k);

    for(int i = 0 ; i < n ; i++) {
        scanf("%d", &arr[i]);
    }

    int count = 0;
    for(int j = 0 ; j < n ; j++) {
        for(int i = 0 ; i < j; i++) {
            if((arr[i] + arr[j]) % k == 0) {
                count++;
            }
        }
    }
    printf("%d\n", count);
    return 0;
}
