    return n && ((n & (n-1)) == 0);

int leftmost_set_bit(unsigned ll int n) {
    int x = 0;
    for(int i = 0 ; n ; i++) {
        if(n & 1) {
            x = i + 1;
        }
        n >>= 1;
    }
    return x;
}
