    }

int main(int argc, char *argv[]) {
    int t;
    scanf("%d", &t);

    while(t--) {
        int n, k;
        scanf("%d%d", &n, &k);
        ll int * counts = (ll int *) calloc(100, sizeof(ll int));
        counts[0] = 1;

        ll int sum = 0;
        for(int i = 0 ; i < n ; i++) {
            int x;
            scanf("%d", &x);
            sum += x;
            if(sum >= k) {
                sum %= k;
            }
            counts[sum]++;
        }

        ll int count = 0;
        for(int i = 0 ; i < k; i++) {
            count += ((counts[i] * (counts[i] - 1)) / 2);
        }
        printf("%lld\n", count);

        free(counts);
    }

    return 0;
}
