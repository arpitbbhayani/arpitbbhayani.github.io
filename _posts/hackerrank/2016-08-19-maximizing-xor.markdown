int readline(char *str) {
    int i = 0;
    char ch;
    while((ch = getchar()) != '\n') {
        str[i++] = ch;
    }
    str[i] = '\0';

int max(int a, int b) {
    return a > b ? a : b;
}

int main(int argc, char *argv[]) {
    int a, b, ans = 0;
    scanf("%d%d", &a, &b);
    for(int i = a; i <= b; i++) {
        for(int j = a; j <= b; j++) {
            ans = max(ans, i ^ j);
        }
    }
    printf("%d\n", ans);
    return 0;
}
