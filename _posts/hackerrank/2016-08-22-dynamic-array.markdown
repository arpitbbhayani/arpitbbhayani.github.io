int main(int argc, char *argv[]) {
    int n, t, lastAns = 0;
    scanf("%d%d", &n, &t);

    v = new vector<int>[n];
    while(t--) {
        int a, x, y;
        scanf("%d%d%d", &a, &x, &y);

        if(a == 1) {
            v[(x ^ lastAns)% n].push_back(y);
        }
        else {
            vector<int> c = v[(x ^ lastAns)% n];
            lastAns = c[y % c.size()];
            cout << lastAns << endl;
        }
    }

    delete []v;
    return 0;
}
