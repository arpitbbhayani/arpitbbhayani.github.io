    int t;

    while( t-- ) {
        int n;
        scanf("%d", &n);
        for(int i = 0 ; i < 100010; i++ ) {
            array[i] = 0;
        }

        for(int i = 0 ; i < n ; i++ ) {
            int x;
            scanf("%d", &x);
            array[x] ++;
        }

        for(int i = 0 ; i < 100010; i++ ) {

            }
        }
    }
    return 0;
